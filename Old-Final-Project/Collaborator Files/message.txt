<script>

let shopperCollection = [];

document.getElementById("shopperForm").addEventListener("submit", function(event) {

  event.preventDefault();

  let isValid = true;

  let email = document.getElementById("email").value.trim();
  let name = document.getElementById("name").value.trim();
  let phone = document.getElementById("phone").value.trim();
  let age = document.getElementById("age").value.trim();
  let address = document.getElementById("address").value.trim();

  document.getElementById("emailError").innerHTML = "";
  document.getElementById("nameError").innerHTML = "";
  document.getElementById("ageError").innerHTML = "";
  document.getElementById("addressError").innerHTML = "";

  if (email === "" || !email.includes("@")) {
    document.getElementById("emailError").innerHTML = "Valid email is required.";
    isValid = false;
  }

  if (name === "") {
    document.getElementById("nameError").innerHTML = "Name is required.";
    isValid = false;
  }

  if (age === "" || isNaN(age) || age <= 0) {
    document.getElementById("ageError").innerHTML = "Valid age is required.";
    isValid = false;
  }

  if (address === "") {
    document.getElementById("addressError").innerHTML = "Address is required.";
    isValid = false;
  }

  if (isValid) {

    let shopper = {
      shopperEmail: email,
      shopperName: name,
      shopperPhone: phone,
      shopperAge: age,
      shopperAddress: address
    };

    shopperCollection.push(shopper);

    document.getElementById("jsonOutput").innerHTML =
      "<strong>JSON Shopper Collection:</strong><br><br>" +
      JSON.stringify(shopperCollection, null, 4);

    document.getElementById("shopperForm").reset();
  }

});

</script>
